1function [AberLev,ValsLev,Inter,SeqFin,LocMax,MaxVal,TxData,EoutCorrD,EoutCorr2,EoutCorr,BerOOK,LocMax2,SeqFin2,MaxVal2,Inter2] = RedDowOok(Ix,TxDataMat,NPPB,ThisCarr,Nb,NumBitDesc,SyncPeriod,CurTesSiz,EyeOpenLow,EyeOpen)
  2
  3PosIx = NPPB/2:NPPB:length(Ix);                                %Possition of the central samples - but the number of samples per symbol is even ????
  4IxAux = Ix(PosIx);                                             %From the main received signal just a few samples are taken for further evaluation
  5n     = 100;                                                   %The number of boxes to be filled up on the histogram process
  6MinDist = n/3;
  7Perce = 0.5;
  8%At this process each eye will be evaluated separetelly. It is
  9%important to remember that the PAM4 has 4 levels, which means
 10%three level of decissions that are exaclty the center of the
 11%eye diagram.
 12Interval = linspace(Perce*(min(Ix)),Perce*(max(Ix)),n);                     %Building the histogram boxes
 13
 14%Therefore, the MATLAB hist function returns the number of
 15%occurrence of each interval.
 16EyeMax = hist(Ix,Interval);
 17EyeMaxaux = [0 EyeMax 0];                                      %Zeros are added at the EyeMax to auxiliate the finding peaks process
 18[~,EyeLoc] = findpeaks(EyeMaxaux,'MinPeakDistance',MinDist,'SortStr','descend','NPeaks',2);%The peaks on the Eye profile will be the levels at the Eyes limit
 19
 20if length(EyeLoc)<2%If it was not able to find the eye profile.
 21    [~,EyeLoc] = findpeaks(EyeMaxaux,'MinPeakDistance',MinDist/2,'SortStr','descend','NPeaks',2);%The peaks on the Eye profile will be the levels at the Eyes limit
 22end
 23
 24if length(EyeLoc)<2%If it was not able to find the eye profile.
 25    %EyeLoc = [2 3];
 26    Levels = [0 -0.2 0.35];
 27else%Whereas, if there is an profile the decission can be found
 28    Levels = [Interval(EyeLoc(1)-1) Interval(EyeLoc(2)-1)];
 29    Levels = sort(Levels);
 30    %                 LevelDec1 = Levels(1) + (Levels(2)-Levels(1))/2 ;
 31    %                 LevelDec2 = Levels(2) + (Levels(3)-Levels(2))/2 ;
 32    %                 LevelDec3 = Levels(3) + (Levels(4)-Levels(3))/2 ;
 33end
 34IxAuxAB = IxAux((IxAux<=Levels(2))&(IxAux>=Levels(1)));
 35Inter = linspace(Levels(1),Levels(2),n);                     %Building the histogram boxes
 36Inter2 = linspace(min(IxAux),max(IxAux),n);                     %Building the histogram boxes
 37EyeAB = hist(IxAuxAB,Inter);                                 %filling up the boxes with samples that fit on them.
 38EyeCD = hist(IxAux,Inter2);                                 %filling up the boxes with samples that fit on them.
 39
 40%What we are looking for here are not where exist the
 41%occurrences of values. The eye is where there is not samples,
 42%this means, the decission level is a reagion between actual
 43%levels thus this reagion does not contain any signal.
 44EyeAB = ~EyeAB;                                                %Changing zeros to one - Zeros compose the eye region
 45EyeCD = ~EyeCD;                                                %Changing zeros to one - Zeros compose the eye region
 46%Starting variables that will be used to identify the eye
 47%diagram.
 48Count   = 1;
 49Count2   = 1;
 50
 51SeqOnes  = zeros(1,length(EyeAB));
 52SeqOnes2  = zeros(1,length(EyeAB));
 53
 54SeqIni  = 1;
 55SeqIni2  = 1;
 56
 57SeqFin  = zeros(1,length(EyeAB));
 58SeqFin2  = zeros(1,length(EyeAB));
 59%The for loop will take account of every box with ones. It is
 60%important to take note that the not operator was used in this
 61%vector, therefore ones means zeros (the eye diagram -
 62%possibly) and zeros means values abouve zeroa (not the eye).
 63for kk=1:length(EyeAB)                                         %For every box
 64    if EyeAB(kk)                                               %if it contains "1"
 65        SeqOnes(SeqIni)=Count;                           %count this element as part of a consecutive sequency
 66        Count = Count + 1;                                 %adds one to the counter of consecutive elements "1"
 67        if kk==length(EyeAB)                                   %if the current box is the last box we got to an end
 68            SeqFin(SeqIni) = kk;                           %The final sequency element is equal to its possition (kk)
 69        end
 70    else                                                       %else if the current box contains "0"
 71        SeqFin(SeqIni) = kk-1;                             %the previous element was the last element of a consecutive sequency of ones
 72        SeqIni = SeqIni + 1;                               %adds one to the counter of consecutive number (take in account how many sequencies there are)
 73        Count = 1;                                           %reset the counter of consecutive elements to mark the start of a new consecutive sequence
 74    end
 75    
 76    if EyeCD(kk)                                               %if it contains "1"
 77        SeqOnes2(SeqIni2)=Count2;                           %count this element as part of a consecutive sequency
 78        Count2 = Count2 + 1;                                 %adds one to the counter of consecutive elements "1"
 79        if kk==length(EyeCD)                                   %if the current box is the last box we got to an end
 80            SeqFin2(SeqIni2) = kk;                           %The final sequency element is equal to its possition (kk)
 81        end
 82    else                                                       %else if the current box contains "0"
 83        SeqFin2(SeqIni2) = kk-1;                             %the previous element was the last element of a consecutive sequency of ones
 84        SeqIni2 = SeqIni2 + 1;                               %adds one to the counter of consecutive number (take in account how many sequencies there are)
 85        Count2 = 1;                                           %reset the counter of consecutive elements to mark the start of a new consecutive sequence
 86    end
 87end
 88
 89
 90%If the eye is open, which means there is a clear difference
 91%between adjacent levels, the eye diagram will be the longest
 92%sequence of ones.
 93[MaxVal,LocMax]=max(SeqOnes);
 94if LocMax<2 || MaxVal<2                                    %if any sequency was found or there is just one sequency it is a error thus
 95    LevDec = 0.0;                                            %the decission level will be by default 0.67. Also, the other variables will
 96    LocMax = 1;                                              %will be set with values to not cause errors in the future
 97    SeqFin(1)=2;
 98    MaxVal = 0;
 99    Inter(1)=LevDec;
100else                                                           %if a sequency was found the middle of the eye will be the middle of the sequency
101    if (SeqFin(LocMax)-MaxVal/2)<1                       %if for some reason the final element of the sequency minus the half of its
102        LevDec = 0.0;                                         %length results in a negative value, something went very wrong, and by
103    else                                                       %default it will be set to 0.7
104        LevDec = Inter(round(SeqFin(LocMax)-MaxVal/2));%Otherwise, the decission level is the middle of the sequency
105    end
106end
107[MaxVal2,LocMax2]=max(SeqOnes2);
108if LocMax2<2 || MaxVal2<2                                    %if any sequency was found or there is just one sequency it is a error thus
109    LevDec2 = 0.0;                                            %the decission level will be by default 0.67. Also, the other variables will
110    LocMax2 = 1;                                              %will be set with values to not cause errors in the future
111    SeqFin2(1)=2;
112    MaxVal2 = 0;
113    Inter2(1)=LevDec2;
114else                                                           %if a sequency was found the middle of the eye will be the middle of the sequency
115    if (SeqFin2(LocMax2)-MaxVal2/2)<1                       %if for some reason the final element of the sequency minus the half of its
116        LevDec2 = 0.0;                                         %length results in a negative value, something went very wrong, and by
117    else                                                       %default it will be set to 0.7
118        LevDec2 = Inter2(round(SeqFin2(LocMax2)-MaxVal2/2));%Otherwise, the decission level is the middle of the sequency
119    end
120end
121
122%another way to measure the eye opening is the get all the
123%boxes and find all peaks on it, that will be a plato created
124%by the sequences of ones (which was zeros). From thos peaks,
125%the eye diagram will be the longer of them hence it will take
126%the most part of the vector that store the findpeaks result.
127%Thus, the middle of the eye will be basically the middle of
128%the peaks vector.
129Loc = find(EyeAB);
130if ~any(Loc)                                              %if for some reason there are no peaks, something went wrong.
131    Loc = 1;
132    LevelDec = 0.0;%mean(Levels(3:4));                       %by default the decission level will be set to 0.65
133else
134    LevelDec = LevDec;
135end
136Loc2 = find(EyeCD);
137if ~any(Loc2)                                              %if for some reason there are no peaks, something went wrong.
138    Loc2 = 1;
139    LevelDec2 = 0.0;%mean(Levels(3:4));                       %by default the decission level will be set to 0.65
140else
141    LevelDec2 = LevDec2;
142end
143%##########################################################################
144
145AberLev = Inter(SeqFin(LocMax))-Inter(round(SeqFin(LocMax)-MaxVal));
146ValsLev = LevDec;
147%%         Finding Decission Levels
148%The process for decoding the income signal will be based on
149%eletronic comparators. Inasmuch as the right decission level
150%must be acquired for accurately decide, within a symbol
151%periode, what that current leavel means (ones or zeros).
152%
153%The process hereafter of chosing the  decission levels is not
154%deterministic rather it is a statistic process. The main idea
155%is to take the decission level from the histogram generated
156%from the income signal stream.
157%
158%This process is realized inside the function Olho_mex.
159%
160%Basicaly the decission level will be the minimal value of the
161%currente eye under evaluation plus the half of the its eye
162%opening.The following ilustration better describe this process
163%
164%Eye Limits:   + (1/2 * Eye Opening:)  =    Comparisson Limit:
165%
166%UperLevel  ______     ______________     _____
167%                 \   /      |       \   /
168%                  \ /       |        \ /
169%                   \ Half Eye Opening /     Decission Level
170%                  / \       |        / \
171%LowerLevel ______/   \______|_______/   \_____
172%
173%
174%%      Actualy Receiving Data
175%Once the signal was processed the next step is through a
176%comparator decide the actual information received.
177%ThisDataPos = 1:NPPB:length(Ix);
178ThisDataSize = NPPB/2:NPPB:length(Ix);
179EoutCorr = zeros(1,length(ThisDataSize));                                                 %Initialization of the vector that will store the income data
180EoutCorrD = zeros(1,length(ThisDataSize));                                                 %Initialization of the vector that will store the income data
181EoutCorr2 = zeros(1,length(ThisDataSize));                                                 %Initialization of the vector that will store the income data
182for kk=1:length(ThisDataSize)%length(Ix(ThisDataSize))                                       %The comparison process will be made for each symbol period
183    %An small portion of the income signal is take for
184    %evaluation by measuring the avarage value of the samples
185    %taken
186    %                 CalcMean = mean((Ix((kk-1)+SymLoc(1))));
187    CalcMean = mean((Ix((kk-1)*NPPB+NPPB/2)));
188    %Verifying the interval for each symbol received.
189    if CalcMean >= LevDec%EyeOpenLow+EyeOpen/2                        %If it is the uper level the incoming data
190        EoutCorr(kk) = 1;                               %is 1
191    else                                                       %If it is the lowest level the incoming data
192        EoutCorr(kk) = 0;                               %is 0
193    end
194    CalcMean2 = mean((Ix((kk-1)*NPPB+NPPB/2)));
195    %Verifying the interval for each symbol received.
196    if CalcMean2 >= LevDec2%EyeOpenLow+EyeOpen/2                        %If it is the uper level the incoming data
197        EoutCorr2(kk) = 1;                               %is 1
198    else                                                       %If it is the lowest level the incoming data
199        EoutCorr2(kk) = 0;                               %is 0
200    end
201    if CalcMean2 >= EyeOpenLow+EyeOpen/2                        %If it is the uper level the incoming data
202        EoutCorrD(kk) = 1;                               %is 1
203    else                                                       %If it is the lowest level the incoming data
204        EoutCorrD(kk) = 0;                               %is 0
205    end
206end
207
208TxDataA = TxDataMat(ThisCarr,:);
209TxDataB = reshape(TxDataA,(Nb-NumBitDesc),CurTesSiz);
210TxDataC = TxDataB(1+SyncPeriod:end-SyncPeriod,:);
211TxData  = reshape(TxDataC,1,size(TxDataC,1)*size(TxDataC,2));
212%%       Calculating the Bit Error Ratio (BER)
213%The final process here is to count the number of wrongdoings
214%of this whole process upon the transmited data for
215%quantitative analizes
216BitErr = sum(xor(TxData,EoutCorr));%Comparison between the Transmited and received and counting the differences
217BitErr2 = sum(xor(TxData,EoutCorr2));%Comparison between the Transmited and received and counting the differences
218BitErrD = sum(xor(TxData,EoutCorrD));%Comparison between the Transmited and received and counting the differences
219if BitErr2<=BitErr
220    if BitErr2<=BitErrD
221        BerOOK = BitErr2/(((Nb-NumBitDesc)-2*SyncPeriod)*CurTesSiz);%Calculating the ration of wrong bits and the total number of bits transmited
222        AberLev = Inter2(SeqFin2(LocMax2))-Inter2(round(SeqFin2(LocMax2)-MaxVal2));
223        ValsLev = LevDec2;
224    else
225        BerOOK  = BitErrD/(((Nb-NumBitDesc)-2*SyncPeriod)*CurTesSiz);%Calculating the ration of wrong bits and the total number of bits transmited
226        AberLev = EyeOpen;
227        ValsLev = EyeOpenLow+EyeOpen/2;
228    end
229else
230    if BitErr<=BitErrD
231        BerOOK = BitErr/(((Nb-NumBitDesc)-2*SyncPeriod)*CurTesSiz);%Calculating the ration of wrong bits and the total number of bits transmited
232    else
233        BerOOK  = BitErrD/(((Nb-NumBitDesc)-2*SyncPeriod)*CurTesSiz);%Calculating the ration of wrong bits and the total number of bits transmited
234        AberLev = EyeOpen;
235        ValsLev = EyeOpenLow+EyeOpen/2;
236    end
237end
238end
